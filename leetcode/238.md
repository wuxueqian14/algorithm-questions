# 238. 除自身以外数组的乘积

## 前后缀

时间复杂度：O(n)。

空间复杂度：O(n)。

```cpp
class Solution {
public:
    vector<int> productExceptSelf(vector<int>& nums) {
        int size = nums.size();
        vector<int> pre(size, 1), nxt(size, 1);
        for (int i = 1; i < size; i++) {
            pre[i] = pre[i-1] * nums[i-1];
        }
        for (int i = size-2; i >= 0; i--) {
            nxt[i] = nxt[i+1] * nums[i+1];
        }
        vector<int> ans(size);
        for (int i = 0; i < size; i++) {
            ans[i] = pre[i] * nxt[i];
        }
        return ans;
    }
};
```
