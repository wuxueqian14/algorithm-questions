# 322. 零钱兑换

## 记忆化搜索

时间复杂度：O(n * amount)。

空间复杂度：O(n * amount)。

```cpp
class Solution {
private:
    int cache[12][10001];
public:
    int coinChange(vector<int>& coins, int amount) {
        int size = coins.size();
        memset(cache, -1, sizeof(cache));
        int ans = dfs(coins, size-1, amount);
        if (ans < 1e5) {
            return ans;
        }
        return -1;
    }

    int dfs(vector<int>& coins, int i, int c) {
        if (i < 0) {
            if (c == 0) {
                return 0;
            } else {
                return 1e5;
            }
        }
        if (cache[i][c] != -1) {
            return cache[i][c];
        }
        if (c < coins[i]) {
            return dfs(coins, i-1, c);
        }
        int ret = min(dfs(coins, i-1, c), dfs(coins, i, c-coins[i])+1);
        cache[i][c] = ret;
        return ret;
    }
};
```

## DP

时间复杂度：O(n * amount)。

空间复杂度：O(n * amount)。

```cpp
class Solution {
public:
    int coinChange(vector<int>& coins, int amount) {
        int size = coins.size();
        int dp[size+1][amount+1];
        dp[0][0] = 0;
        for (int j = 1; j <= amount; j++) {
            dp[0][j] = 1e5;
        }
        for (int i = 0; i < size; i++) {
            for (int j = 0; j <= amount; j++) {
                if (j < coins[i]) {
                    dp[i+1][j] = dp[i][j];
                } else {
                    dp[i+1][j] = min(dp[i][j], dp[i+1][j-coins[i]]+1);
                }
            }
        }
        if (dp[size][amount] < 1e5) {
            return dp[size][amount];
        }
        return -1;
    }
};
```

## 优化

一个数组，正向计算。

时间复杂度：O(n * amount)。

空间复杂度：O(amount)。

```cpp
class Solution {
public:
    int coinChange(vector<int>& coins, int amount) {
        int size = coins.size();
        int dp[amount+1];
        dp[0] = 0;
        for (int j = 1; j <= amount; j++) {
            dp[j] = 1e5;
        }
        for (int i = 0; i < size; i++) {
            for (int j = 0; j <= amount; j++) {
                if (j >= coins[i]) {
                    dp[j] = min(dp[j], dp[j-coins[i]]+1);
                }
            }
        }
        if (dp[amount] < 1e5) {
            return dp[amount];
        }
        return -1;
    }
};
```
