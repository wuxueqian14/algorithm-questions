# 322. 零钱兑换

## 完全背包

时间复杂度：O(n * amount)。

空间复杂度：O(n * amount)。

```cpp
class Solution {
public:
    int coinChange(vector<int>& coins, int amount) {
        int size = coins.size();
        int dp[size+1][amount+1];
        dp[0][0] = 0;
        for (int i = 1; i <= amount; i++) {
            dp[0][i] = 1e5;
        }
        for (int i = 0; i < size; i++) {
            for (int j = 0; j <= amount; j++) {
                if (j >= coins[i]) {
                    dp[i+1][j] = min(dp[i][j], dp[i+1][j-coins[i]]+1);
                } else {
                    dp[i+1][j] = dp[i][j];
                }
            }
        }
        if (dp[size][amount] != 1e5) {
            return dp[size][amount];
        }
        return -1;
    }
};
```

## 优化

一个数组，正向计算。

时间复杂度：O(n * amount)。

空间复杂度：O(amount)。

```cpp
class Solution {
public:
    int coinChange(vector<int>& coins, int amount) {
        int size = coins.size();
        int dp[amount+1];
        dp[0] = 0;
        for (int i = 1; i <= amount; i++) {
            dp[i] = 1e5;
        }
        for (int i = 0; i < size; i++) {
            for (int j = 0; j <= amount; j++) {
                if (j >= coins[i]) {
                    dp[j] = min(dp[j], dp[j-coins[i]]+1);
                } 
            }
        }
        if (dp[amount] != 1e5) {
            return dp[amount];
        }
        return -1;
    }
};
```
