# 215. 数组中的第K个最大元素

## 快速排序

每次分区可以确定一个元素的位置，分区后判断位置与k位置的大小关系，递归子区间。

时间复杂度：O(n)。

```cpp
class Solution {
public:
    int findKthLargest(vector<int>& nums, int k) {
        int len = nums.size();
        int l = 0, r = len - 1, t = len - k;
        while (true) {
            int idx = partition(nums, l, r);
            if (idx == t) {
                return nums[idx];
            } else if (idx > t) {
                r = idx - 1;
            } else {
                l = idx + 1;
            }
        }
        return -1;
    }

    int partition(vector<int>& nums, int l, int r) {
        int i = l + rand() % (r-l+1);
        swap(nums[l], nums[i]);
        int pivot = nums[l];
        while (l < r) {
            while (l < r && nums[r] >= pivot) {
                r--;
            }
            swap(nums[l], nums[r]);
            while (l < r && nums[l] <= pivot) {
                l++;
            }
            swap(nums[l], nums[r]);
        }
        return l;
    }
};
```
