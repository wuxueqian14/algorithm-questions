# 2. Add Two Numbers

## 思路

和合并链表思路类似，从低位开始加，记录每次进位，修改指向，当p1和p2长度不相等时，继续单循环，最后判断是否需要进位，进位需要添加结点。

时间复杂度：O(n)。

```c
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* addTwoNumbers(ListNode* l1, ListNode* l2) {
        ListNode *h = new ListNode(0), *cur = h, *s;
        int t, add = 0;
        while (l1 || l2) {
            t = 0;
            if (l1) {
                t += l1->val;
                l1 = l1->next;
            }
            if (l2) {
                t += l2->val;
                l2 = l2->next;
            }
            t += add;
            add = t / 10;
            s = new ListNode(t%10);
            cur->next = s;
            cur = cur->next;
        }
        if (add) {
            s = new ListNode(1);
            cur->next = s;
            cur = cur->next;
        }
        return h->next;
    }
};
```
