# 51. N 皇后

## 排列型回溯

模拟每一行。

```cpp
class Solution {
private:
    int n;
    vector<vector<string>> ans;
public:
    vector<vector<string>> solveNQueens(int n) {
        this->n = n;
        vector<string> v(n, string(n, '.'));
        backtrack(v, 0);
        return ans;
    }

    void backtrack(vector<string> &v, int i) {
        if (i == n) {
            ans.push_back(v);
            return;
        }
        for (int j = 0; j < n; j++) {
            if (check(v, i, j)) {
                v[i][j] = 'Q';
                backtrack(v, i+1);
                v[i][j] = '.';
            }
        }
    }

    bool check(vector<string> &v, int r, int c) {   
        for (int i = 0; i < n; i++) {
            if (v[i][c] == 'Q') {
                return false;
            }
        }
        for (int i = r-1, j = c-1; i >= 0 && j >= 0; i--, j--) {
            if (v[i][j] == 'Q') {
                return false;
            }
        }
        for (int i = r-1, j = c+1; i >= 0 && j < n; i--, j++) {
            if (v[i][j] == 'Q') {
                return false;
            }
        }
        return true;
    }
};
```
