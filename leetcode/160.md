# 160. 相交链表 

## 链表

时间复杂度：O(n + m)。

```cpp
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode *getIntersectionNode(ListNode *headA, ListNode *headB) {
        int len1 = getLength(headA), len2 = getLength(headB);
        ListNode *l1, *l2;
        int step = 0;
        if (len1 > len2) {
            step = len1 - len2;
            l1 = headA;
            l2 = headB;
        } else {
            step = len2 - len1;
            l1 = headB;
            l2 = headA;
        }
        for (int i = 0; i < step; i++) {
            l1 = l1->next;
        }
        while (l1 && l2) {
            if (l1 == l2) {
                return l1;
            }
            l1 = l1->next;
            l2 = l2->next;
        }
        return NULL;
    }

    int getLength(ListNode *head) {
        int ret = 0;
        while (head) {
            ret++;
            head = head->next;
        }
        return ret;
    }
};
```
