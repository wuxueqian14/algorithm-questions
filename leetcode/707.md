# 707. 设计链表

熟悉单链表的增删改。

```cpp
class MyLinkedList {
private:
    typedef struct Node {
        int val;
        Node *next;
    } Node, *LinkList;

    LinkList head;
    int size;
public:
    MyLinkedList() {
        head = new Node();
        head->next = nullptr;
        size = 0;
    }
    
    int get(int index) {
        if (index < 0 || index > size - 1) {
            return -1;
        }
        LinkList p = head->next;
        int i = 0;
        while (i < index) {
            p = p->next;
            i++;
        }
        return p->val;
    }
    
    void addAtHead(int val) {
        LinkList t = new Node();
        t->val = val;
        t->next = head->next;
        head->next = t;
        size++;
    }
    
    void addAtTail(int val) {
        LinkList p = head;
        while (p->next) {
            p = p->next;
        }
        LinkList t = new Node();
        t->next = nullptr;
        t->val = val;
        p->next = t;
        size++;
    }
    
    void addAtIndex(int index, int val) {
        if (index == size) {
            addAtTail(val);
        } else if (index > size) {
            return;
        } else {
            int i = 0;
            LinkList p = head;
            while (i < index) {
                p = p->next;
                i++;
            }
            LinkList t= new Node();
            t->val = val;
            t->next = p->next;
            p->next = t;
            size++;
        }
    }
    
    void deleteAtIndex(int index) {
        if (index < 0 || index > size - 1) {
            return;
        }
        LinkList p = head;
        int i = 0;
        while (i < index) {
            p = p->next;
            i++;
        }
        p->next = p->next->next;
        size--;
    }
};

/**
 * Your MyLinkedList object will be instantiated and called as such:
 * MyLinkedList* obj = new MyLinkedList();
 * int param_1 = obj->get(index);
 * obj->addAtHead(val);
 * obj->addAtTail(val);
 * obj->addAtIndex(index,val);
 * obj->deleteAtIndex(index);
 */
```
