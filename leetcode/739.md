# 739. 每日温度

## 单调栈

及时去掉无用数据，保证栈中数据有序。

### 从前往后

时间复杂度：O(n)。

```cpp
class Solution {
public:
    vector<int> dailyTemperatures(vector<int>& temperatures) {
        stack<int> s;
        int size = temperatures.size();
        vector<int> ans(size, 0);
        for (int i = 0; i < size; i++) {
            while (!s.empty() && temperatures[i] > temperatures[s.top()]) {
                int cur = s.top();
                ans[cur] = i - cur;
                s.pop();
            }
            s.push(i);
        }
        return ans;
    }
};
```

### 从后往前

时间复杂度：O(n)。

```cpp
class Solution {
public:
    vector<int> dailyTemperatures(vector<int>& temperatures) {
        stack<int> s;
        int size = temperatures.size();
        vector<int> ans(size, 0);
        for (int i = size - 1; i >= 0; i--) {
            while (!s.empty() && temperatures[i] >= temperatures[s.top()]) {
                s.pop();
            }
            if (!s.empty()) {
                ans[i] = s.top() - i;
            }
            s.push(i);
        }
        return ans;
    }
};
```
