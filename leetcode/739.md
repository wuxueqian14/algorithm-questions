# 739. 每日温度

## 单调栈

### 从前往后

时间复杂度：O(n)。

```cpp
class Solution {
private:
    stack<int> s;
public:
    vector<int> dailyTemperatures(vector<int>& temperatures) {
        int size = temperatures.size();
        vector<int> ans(size, 0);
        for (int i = 0; i < size; i++) {
            while (!s.empty() && temperatures[i] > temperatures[s.top()]) {
                int j = s.top();
                ans[j] = i-j;
                s.pop();
            }
            s.push(i);
        }
        return ans;
    }
};
```

### 从后往前

时间复杂度：O(n)。

```cpp
class Solution {
private:
    stack<int> s;
public:
    vector<int> dailyTemperatures(vector<int>& temperatures) {
        int size = temperatures.size();
        vector<int> ans(size, 0);
        for (int i = size-1; i >= 0; i--) {
            while (!s.empty() && temperatures[i] >= temperatures[s.top()]) {
                s.pop();
            }
            if (!s.empty()) {
                ans[i] = s.top() - i;
            }
            s.push(i);
        }
        return ans;
    }
};
```
