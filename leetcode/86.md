# 86. 分隔链表

## 暴力

用两个链表维护比x小，大于等于x。每遍历一个结点添加到相应的链表中。

时间复杂度：O(n)。

```cpp
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* partition(ListNode* head, int x) {
        ListNode *l1 = new ListNode(), *l2 = new ListNode();
        ListNode *p1 = head, *p2 = l1, *p3 = l2;
        while (p1) {
            if (p1->val < x) {
                p2->next = p1;
                p2 = p2->next;
            } else {
                p3->next = p1;
                p3 = p3->next;
            }
            p1 = p1->next;
        }
        p3->next = nullptr;
        p2->next = l2->next;
        return l1->next;
    }
};
```
