# 86. Partition List

## 暴力

用两个链表维护比x小，大于等于x。每遍历一个结点添加到相应的链表中。

时间复杂度：O(n)。

```c
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* partition(ListNode* head, int x) {
        ListNode *smallHead = new ListNode();
        smallHead->next = nullptr;
        ListNode *largeHead = new ListNode();
        largeHead->next = nullptr;
        ListNode *small = smallHead, *large = largeHead;
        while (head) {
            if (head->val < x) {
                small->next = head;
                small = head;
            } else {
                large->next = head;
                large = head;
            }
            head = head->next;
        }
        large->next = nullptr;
        small->next = largeHead->next;
        return smallHead->next;
    }
};
```
