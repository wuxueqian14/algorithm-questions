# 131. 分割回文串

## 子集型回溯

模拟逗号选还是不选，最后一个字符必须选。

时间复杂度：O(n * 2<sup>n</sup>)

```cpp
class Solution {
private:
    vector<vector<string>> ans;
    vector<string> v;
public:
    vector<vector<string>> partition(string s) {
        backtrack(s, 0, 0);
        return ans;
    }

    void backtrack(string &s, int i, int start) {
        if (i == s.size()) {
            ans.push_back(v);
            return;
        }
        if (check(s, start, i)) {
            // 选，构成子串
            v.push_back(s.substr(start, i-start+1));
            backtrack(s, i+1, i+1);
            v.pop_back();
        }
        if (i < s.size() - 1) {
            // 不选，继续添加
            backtrack(s, i+1, start);
        }
    }

    bool check(string &t, int i, int j) {
        while (i < j) {
            if (t[i] != t[j]) {
                return false;
            }
            i++, j--;
        }
        return true;
    }
};
```

## 枚举子串结束的位置

```cpp
class Solution {
private:
    vector<vector<string>> ans;
    vector<string> v;
public:
    vector<vector<string>> partition(string s) {
        backtrack(s, 0);
        return ans;
    }

    void backtrack(string &s, int i) {
        if (i == s.size()) {
            ans.push_back(v);
            return;
        }
        for (int j = i; j < s.size(); j++) {
            if (check(s, i, j)) {
                v.push_back(s.substr(i, j-i+1));
                backtrack(s, j+1);
                v.pop_back();
            }
        }
    }

    bool check(string &t, int i, int j) {
        while (i < j) {
            if (t[i] != t[j]) {
                return false;
            }
            i++, j--;
        }
        return true;
    }
};
```
