# 143. 重排链表

## 反转链表+寻找中间值+合并链表

时间复杂度：O(n)。

```java
/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode() {}
 *     ListNode(int val) { this.val = val; }
 *     ListNode(int val, ListNode next) { this.val = val; this.next = next; }
 * }
 */
class Solution {
    private ListNode reverseList(ListNode head) {
        ListNode p = head, q = head.next, t;
        head.next = null;
        while (q != null) {
            t = q.next;
            q.next = p;
            p = q;
            q = t; 
        }
        return p;
    }

    public void reorderList(ListNode head) {
        if (head.next == null) {
            return;
        }
        ListNode s = head, f = head;
        while (f != null && f.next != null) {
            f = f.next.next;
            s = s.next;
        }
        ListNode p = head;
        while (p.next != s) {
            p = p.next;
        }
        p.next = null;
        ListNode l1 = head, l2 = reverseList(s);
        mergeList(l1, l2);
    }

    private void mergeList(ListNode l1, ListNode l2) {
        ListNode head = new ListNode(), p = head;
        while (l1 != null && l2 != null) {
            p.next = l1;
            l1 = l1.next;
            p = p.next;

            p.next = l2;
            l2 = l2.next;
            p = p.next;
        }
        while (l1 != null) {
            p.next = l1;
            p = p.next;
            l1 = l1.next;
        }
        while (l2 != null) {
            p.next = l2;
            p = p.next;
            l2 = l2.next;
        }
    }
}
```
