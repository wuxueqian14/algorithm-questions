# 33. 搜索旋转排序数组

## 二分查找

根据部分有序判断target所在区间。左区间target左侧，右区间target右侧。

时间复杂度：O(logn)。

```cpp
class Solution {
public:
    int search(vector<int>& nums, int target) {
        int l = 0, r = nums.size()-1;
        while (l <= r) {
            int mid = l + (r-l)/2;
            if (nums[mid] == target) {
                return mid;
            } else if (check(nums, mid, target)) {
                l = mid + 1;
            } else {
                r = mid - 1;
            }
        }
        return -1;
    }

    bool check(vector<int>& nums, int mid, int target) {
        int end = nums.size() - 1;
        if (nums[mid] < nums[end]) {
            return nums[mid] < target && target <= nums[end];
        } else {
            return nums[mid] < target || target <= nums[end];
        }
    }
};
```
