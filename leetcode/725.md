# 725. Split Linked List in Parts

## 暴力

先统计链表长度，然后计算每个部分多少个，余数多少，前面多少个部分需要多1个元素。

时间复杂度：O(n)。

```c
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    vector<ListNode*> splitListToParts(ListNode* head, int k) {
        ListNode *p = head, *p1;
        int cnt = 0;
        while (p) {
            p = p->next;
            cnt++;
        }
        vector<ListNode*> v;
        int n = cnt / k, m = cnt % k;
        if (n == 0) {
            n = 1;
            m = 0;
        }
        p = head;
        for (int i = 0; i < k; i++) {
            v.push_back(p);
            // 去找到当前部分的尾部
            for (int j = 1; j < n; j++) {
                p = p->next;
            }
            // 如果有余数，当前部分长度加1
            if (m && p) {
                p = p->next;
                m--;
            }
            if (!p) {
                continue;
            }
            p1 = p->next;
            p->next = nullptr;
            p = p1;
        }
        return v;
    }
};
```
