# 725. 分隔链表

## 暴力

先统计链表长度，然后计算每个部分多少个，余数多少，前面多少个部分需要多1个元素。

时间复杂度：O(n)。

```cpp
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    vector<ListNode*> splitListToParts(ListNode* head, int k) {
        ListNode *p = head;
        int len = 0;
        while (p) {
            len++;
            p = p->next;
        }
        vector<ListNode*> ans;
        int n = len / k, m = len % k, cnt;
        p = head;
        for (int i = 0; i < k; i++) {
            ans.push_back(p);
            if (!p) {
                continue;
            }
            if (m) {
                cnt = n + 1;
                m--;
            } else {
                cnt = n;
            }
            for (int j = 1; j < cnt; j++) {
                p = p->next;
            }
            ListNode *t = p->next;
            p->next = nullptr;
            p = t;
        }
        return ans;
    }
};
```
