# 146. LRU 缓存

# 哈希表+双向链表

时间复杂度：get和put操作都是O(1)。

```cpp
struct Node {
    int key, val;
    struct Node *prev, *next;
    Node() : key(0), val(0), prev(nullptr), next(nullptr) {}
    Node(int key, int val) : key(key), val(val), prev(nullptr), next(nullptr) {}
};

class LRUCache {
private:
    unordered_map<int, Node*> m;
    Node *dummy;
    int capacity;
    void deleteNode(Node *node) {
        node->prev->next = node->next;
        node->next->prev = node->prev;
    }
    void addHead(Node *node) {
        node->next = dummy->next;
        dummy->next->prev = node;
        dummy->next = node;
        node->prev = dummy;
    }
public:
    LRUCache(int capacity) {
        dummy = new Node();
        dummy->next = dummy;
        dummy->next = dummy;
        this->capacity = capacity;
    }
    
    int get(int key) {
        if (!m.count(key)) {
            return -1;
        }
        Node *node = m[key];
        deleteNode(node);
        addHead(node);
        m[key] = dummy->next;
        return node->val;
    }
    
    void put(int key, int value) {
        if (!m.count(key)) {
            if (m.size() == capacity) {
                Node *node = dummy->prev;
                deleteNode(node);
                m.erase(node->key);
            }
            Node *node = new Node(key, value);
            addHead(node);
            m[key] = dummy->next;
        } else {
            Node *node = m[key];
            deleteNode(node);
            node->val = value;
            addHead(node);
            m[key] = dummy->next;
        }
    }
};

/**
 * Your LRUCache object will be instantiated and called as such:
 * LRUCache* obj = new LRUCache(capacity);
 * int param_1 = obj->get(key);
 * obj->put(key,value);
 */
```

### map+list

> list是一个双向链表，在list中任何位置进行插入/删除操作速度都很快，常量时间内完成，插入和删除不会造成迭代器失效。deque是双端队列，头尾两端插入和删除元素比较高效，中间删除和添加元素效率低，在deque两端插入和删除并不会使其它元素的指针或引用失效。


```cpp
class LRUCache {
private:
    list<pair<int,int>> q;
    unordered_map<int, list<pair<int,int>>::iterator> m;
    int size;

    void moveToHead(int key, int value) {
        auto it = m[key];
        pair<int,int> p = *it;
        q.erase(it);
        addHead(p.first, value);
    }

    void addHead(int key, int value) {
        q.push_front(pair<int,int>(key, value));
        m[key] = q.begin();
        if (m.size() > size) {
            moveTail();
        }
    }

    void moveTail() {
        pair<int,int> p = q.back();
        q.pop_back();
        m.erase(p.first);
    }

public:
    LRUCache(int capacity) {
        size = capacity;
    }
    
    int get(int key) {
        if (!m.count(key)) {
            return -1;
        }
        int value = m[key]->second;
        moveToHead(key, value);
        return value;
    }
    
    void put(int key, int value) {
        if (m.count(key)) {
            moveToHead(key, value);
        } else {
            addHead(key, value);
        }
    }
};

/**
 * Your LRUCache object will be instantiated and called as such:
 * LRUCache* obj = new LRUCache(capacity);
 * int param_1 = obj->get(key);
 * obj->put(key,value);
 */
```
