# 977. 有序数组的平方

## 二分查找+双指针

先通过二分查找找到正负数边界，再用双指针指向比较大小添加到ans中。

时间复杂度：O(n)。

```cpp
class Solution {
public:
    vector<int> sortedSquares(vector<int>& nums) {
        vector<int> v;
        int j = findGreaterZero(nums);
        int i = j - 1;
            while (i >= 0 && j < nums.size()) {
                int t1 = nums[i] * nums[i];
                int t2 = nums[j] * nums[j];
                if (t1 < t2) {
                    v.push_back(t1);
                    i--;
                } else {
                    v.push_back(t2);
                    j++;
                }
            }
            while (i >= 0) {
                v.push_back(nums[i] * nums[i]);
                i--;
            }
            while (j < nums.size()) {
                v.push_back(nums[j] * nums[j]);
                j++;
            }
        return v;
    }

    int findGreaterZero(vector<int>& nums) {
        int l = 0, r = nums.size() - 1;
        while (l <= r) {
            int mid = l + (r - l) / 2;
            if (nums[mid] >= 0) {
                r = mid - 1;
            } else {
                l = mid + 1;
            }
        }
        return l;
    }
};
```
