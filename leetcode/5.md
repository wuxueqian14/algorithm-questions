# 5. 最长回文子串

## DP

时间复杂度：O(n<sup>2</sup>)。

```cpp
class Solution {
public:
    string longestPalindrome(string s) {
        int size = s.size();
        bool dp[size][size];
        memset(dp, false, sizeof(dp));
        for (int i = 0; i < size; i++) {
            dp[i][i] = true;
        }
        int start = 0, len = 0;
        for (int j = 0; j < size; j++) {
            for (int i = 0; i <= j; i++) {
                if (s[i] == s[j]) {
                    if (i == j || i + 1 == j) {
                        dp[i][j] = true;
                    } else {
                        dp[i][j] = dp[i+1][j-1];
                    }
                } else {
                    dp[i][j] = false;
                }
                if (dp[i][j]) {
                    if (j-i+1 > len) {
                        len = j-i+1;
                        start = i;
                    }
                }
            }
        }
        return s.substr(start, len);
    }
};
```
