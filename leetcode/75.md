# 75. 颜色分类

## 

```cpp
class Solution {
public:
    void sortColors(vector<int>& nums) {
        int size = nums.size();
        int cnt1 = 0, cnt2 = 0;
        for (int i = 0; i < size; i++) {
            if (nums[i] == 0) {
                cnt1++;
            } else if (nums[i] == 1) {
                cnt2++;
            }
        }
        for (int i = 0; i < size; i++) {
            if (i < cnt1) {
                nums[i] = 0;
            } else if (i < cnt1+cnt2) {
                nums[i] = 1;
            } else {
                nums[i] = 2;
            }
        }
    }
};
```
