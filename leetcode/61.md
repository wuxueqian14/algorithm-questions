# 61. 旋转链表

## 规律

发现k的次数可以很大，只需要对k%len取模后计算即可。找到断开的结点，前一个结点，尾结点并修改指向即可。

时间复杂度：O(n)。

```cpp
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* rotateRight(ListNode* head, int k) {
        if (!head) {
            return nullptr;
        }
        int len = 0;
        ListNode *p = head;
        while (p) {
            p = p->next;
            len++;
        }
        int m = k % len;
        if (m == 0) {
            return head;
        }
        int t = len - m;
        int i = 1;
        p = head;
        while (i < t) {
            p = p->next;
            i++;
        }
        ListNode *l1 = head, *l2 = p->next;
        p->next = nullptr;
        p = l2;
        while (p->next) {
            p = p->next;
        }
        p->next = l1;
        return l2;
    }
};
```
