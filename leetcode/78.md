# 78. 子集

## 子集型回溯

### 添加n个结点，每次选还是不选。

时间复杂度：O(2<sup>n</sup>)。


```cpp
class Solution {
private:
    vector<vector<int>> ans;
    vector<int> v;
public:
    vector<vector<int>> subsets(vector<int>& nums) {
        dfs(nums, 0);
        return ans;
    }

    void dfs(vector<int>& nums, int i) {
        if (i == nums.size()) {
            ans.push_back(v);
            return;
        }
        dfs(nums, i+1);
        v.push_back(nums[i]);
        dfs(nums, i+1);
        v.pop_back();
    }
};
```

### 每次必须选一个数，再选数只能往后选，每个结点都是答案。

```cpp
class Solution {
private:
    vector<vector<int>> ans;
    vector<int> v;
public:
    vector<vector<int>> subsets(vector<int>& nums) {
        dfs(nums, 0);
        return ans;
    }

    void dfs(vector<int>& nums, int i) {
        ans.push_back(v);
        if (i == nums.size()) {
            return;
        }
        for (int j = i; j < nums.size(); j++) {
            v.push_back(nums[j]);
            dfs(nums, j+1);
            v.pop_back();
        }
    }
};
```
