# 78. 子集

## 子集型回溯

时间复杂度：O(2<sup>n</sup>)。

```cpp
class Solution {
private:
    vector<vector<int>> ans;
public:
    vector<vector<int>> subsets(vector<int>& nums) {
        vector<int> v;
        dfs(nums, v, 0);
        return ans;
    }

    void dfs(vector<int>& nums, vector<int> v, int i) {
        if (i == nums.size()) {
            ans.push_back(v);
            return;
        }
        dfs(nums, v, i+1);
        v.push_back(nums[i]);
        dfs(nums, v, i+1);
        v.pop_back();
    }
};
```
