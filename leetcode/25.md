# 25. K 个一组翻转链表

## 翻转链表

部分翻转列表的升级，每k个翻转，更新翻转前的结点P0。

时间复杂度：O(n)。

空间复杂度：O(1)。

```cpp
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* reverseKGroup(ListNode* head, int k) {
        int n = 0;
        ListNode *p = head;
        while (p) {
            n++;
            p = p->next;
        }
        ListNode *dummy = new ListNode(0, head);
        ListNode *p0 = dummy;
        while (n >= k) {
            n -= k;
            ListNode *pre = nullptr, *cur = p0->next, *nxt;
            for (int i = 0; i < k; i++) {
                nxt = cur->next;
                cur->next = pre;
                pre = cur;
                cur = nxt;
            }
            // 存储反转后的末尾结点
            ListNode *t = p0->next;
            p0->next->next = cur;
            p0->next = pre;
            // 更新
            p0 = t;
        }
        return dummy->next;
    }
};
```
