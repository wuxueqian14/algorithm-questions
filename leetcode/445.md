# 445. 两数相加 II

## 栈辅助

与leetcode2相比，链表是反向的。

时间复杂度：O(n)。

```cpp
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* addTwoNumbers(ListNode* l1, ListNode* l2) {
        l1 = reverseList(l1), l2 = reverseList(l2);
        int t, add = 0;
        ListNode *h = new ListNode(), *p = h;
        while (l1 && l2) {
            t = l1->val + l2->val + add;
            add = t / 10;
            t = t % 10;
            p->next = new ListNode(t);
            p = p->next;
            l1 = l1->next;
            l2 = l2->next;
        }
        while (l1) {
            t = l1->val + add;
            add = t / 10;
            t = t % 10;
            p->next = new ListNode(t);
            p = p->next;
            l1 = l1->next;
        }
        while (l2) {
            t = l2->val + add;
            add = t / 10;
            t = t % 10;
            p->next = new ListNode(t);
            p = p->next;
            l2 = l2->next;
        }
        if (add) {
            p->next = new ListNode(1);
            p = p->next;
        }
        return reverseList(h->next);
    }

    ListNode* reverseList(ListNode *l) {
        ListNode *prv = nullptr, *cur = l, *nxt;
        while (cur) {
            nxt = cur->next;
            cur->next = prv;
            prv = cur;
            cur = nxt;
        }
        return prv;
    }
};
```
