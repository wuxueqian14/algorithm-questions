# 39. 组合总和

## 组合型回溯

时间复杂度：O(n * 2<sup>n</sup>)。

### 选还是不选

```cpp
class Solution {
private:
    vector<vector<int>> ans;
    vector<int> v;
    int sum;
public:
    vector<vector<int>> combinationSum(vector<int>& candidates, int target) {
        sum = 0;
        dfs(candidates, 0, target);
        return ans;
    }

    void dfs(vector<int>& candidates, int i, int target) {
        if (i == candidates.size() || sum > target) {
            return;
        }
        if (sum == target) {
            ans.push_back(v);
            return;
        }
        dfs(candidates, i+1, target);
        v.push_back(candidates[i]);
        sum += candidates[i];
        dfs(candidates, i, target);
        sum -= candidates[i];
        v.pop_back();
    }
};
```

### 每次选1个数

```cpp
class Solution {
private:
    vector<vector<int>> ans;
    vector<int> v;
    int sum;
    int target;
    int len;
public:
    vector<vector<int>> combinationSum(vector<int>& candidates, int target) {
        sum = 0;
        this->target = target;
        len = candidates.size();
        dfs(candidates, 0);
        return ans;
    }

    void dfs(vector<int>& candidates, int i) {
        if (sum > target) {
            return;
        }
        if (sum == target) {
            ans.push_back(v);
            return;
        }
        for (int j = i; j < len; j++) {
            v.push_back(candidates[j]);
            sum += candidates[j];
            dfs(candidates, j);
            v.pop_back();
            sum -= candidates[j];
        }
    }
};
```
