# 39. 组合总和

## 组合型回溯

时间复杂度：O(n * 2<sup>n</sup>)。

### 选还是不选

```cpp
class Solution {
private:
    vector<vector<int>> ans;
    vector<int> t;
    int sum = 0, target_;
public:
    vector<vector<int>> combinationSum(vector<int>& candidates, int target) {
        target_ = target;
        backtrack(candidates, 0);
        return ans;
    }

    void backtrack(vector<int>& candidates, int i) {
        if (i == candidates.size()) {
            return;
        }
        if (sum > target_) {
            return;
        }
        if (sum == target_) {
            ans.push_back(t);
            return;
        }
        t.push_back(candidates[i]);
        sum += candidates[i];
        backtrack(candidates, i);
        sum -= candidates[i];
        t.pop_back();
        backtrack(candidates, i+1);
    }
};
```

### 每次选1个数

```cpp
class Solution {
private:
    vector<vector<int>> ans;
    vector<int> v;
    int sum;
public:
    vector<vector<int>> combinationSum(vector<int>& candidates, int target) {
        backtrack(candidates, 0, target);
        return ans;
    }

    void backtrack(vector<int>& candidates, int i, int target) {
        if (sum > target) {
            return;
        }
        if (sum == target) {
            ans.push_back(v);
            return;
        }

        for (int j = i; j < candidates.size(); j++) {
            v.push_back(candidates[j]);
            sum += candidates[j];
            backtrack(candidates, j, target);
            sum -= candidates[j];
            v.pop_back();
        }
    }
};
```
