# 234. 回文链表

## 

找到第n个结点+翻转链表。

时间复杂度：O(n)。

空间复杂度：O(1)。

```cpp
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    bool isPalindrome(ListNode* head) {
        int len = getLength(head);
        int t;
        if (len % 2 == 0) {
            t = len / 2;
        } else {
            t = len / 2 + 1;
        }
        int i = 1;
        ListNode *p = head;
        while (i < t) {
            p = p->next;
            i++;
        }
        ListNode *l2 = p->next, *l1 = head;
        p->next = nullptr;
        l2 = reverse(l2);
        while (l1 && l2) {
            if (l1->val != l2->val) {
                return false;
            }
            l1 = l1->next;
            l2 = l2->next;
        }
        return true;
    }

    int getLength(ListNode *head) {
        int ret = 0;
        while (head) {
            ret++;
            head = head->next;
        }
        return ret;
    }

    ListNode *reverse(ListNode *head) {
        ListNode *prv = nullptr, *cur = head;
        while (cur) {
            ListNode *t = cur->next;
            cur->next = prv;
            prv = cur;
            cur = t;
        }
        return prv;
    }
};
```
